### 第一题
```python
def recv_exact(sock, n: int) -> bytes:
    data = b""
    while len(data) < n:
        chunk = sock.recv(n - len(data))
        if not chunk:
            raise ConnectionError("连接意外关闭")
        data += chunk
    return data
```

### 第二题
```python
def split_file(data, lmin, lmax, seed):
    rng = random.Random(seed)
    total = len(data)
    chunks = []
    pos = 0
    while pos < total:
        remaining = total - pos
        if remaining <= lmax:
            chunks.append(remaining)
            break
        size = rng.randint(lmin, lmax)
        chunks.append(size)
        pos += size
    return chunks
# seed=42, 520B, [50,100] → N≈7~9, 总和=520
```

### 第七题
```python
# GBN Server — 乱序丢弃
if seq != expected_seq:
    send_ack(expected_seq)  # 重发当前ACK，不推进

# SR Server — 乱序缓存
if base <= seq < base + WINDOW:
    recv_buffer[seq] = payload  # 存下来
    send_ack(seq)               # 单独确认这个包
    while base in recv_buffer:  # 向前滑动
        base += 1

# GBN 超时 — 重传整个窗口
except queue.Empty:
    for s in range(base, next_seq):
        packets[s].sent_time = 0
    next_seq = base

# SR 超时 — 只重传超时的那个
for i in range(base, base + WINDOW):
    if not acked[i] and now - send_times[i] > rto:
        send_packet(i)  # 只重传这一个
```
